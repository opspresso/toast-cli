toast.sh

Install

curl -s toast.sh/install-v3 | bash

Usage

~/toaster/toast.sh config REGION ap-northeast-2
~/toaster/toast.sh config BUCKET repo.toast.sh
~/toaster/toast.sh install {package}
~/toaster/toast.sh version {branch}
~/toaster/toast.sh publish bucket
~/toaster/toast.sh publish beanstalk
~/toaster/toast.sh deploy bucket {bucket}
~/toaster/toast.sh deploy beanstalk {env}
